# Import necessary libraries
import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
from statsmodels.tsa.stattools import adfuller
from statsmodels.tsa.arima.model import ARIMA
from statsmodels.graphics.tsaplots import plot_acf, plot_pacf
from google.colab import files
import io

# Step 1: Upload the text file from your computer
print("Please upload your text file containing the M1 money supply data.")
uploaded = files.upload()

# Step 2: Read the uploaded file
# Assuming the file is named 'm1_data.txt' and has the same format as your document
filename = list(uploaded.keys())[0]  # Get the name of the uploaded file
with io.StringIO(uploaded[filename].decode('utf-8')) as f:
    lines = f.readlines()

# Parse the relevant data (skip header lines until TIME and MONEY columns)
data_start = False
time = []
money = []
for line in lines:
    if line.startswith("TIME\tMONEY"):
        data_start = True
        continue
    if data_start:
        try:
            t, m = line.split()
            time.append(int(t))
            money.append(float(m))
        except:
            break  # Stop at the end of the data

# Create a DataFrame
df = pd.DataFrame({'Time': time, 'Money': money})

# Step 3: Convert to a time series with dates
# Time 1 = January 1951, monthly frequency
start_date = pd.to_datetime('1951-01-01')
df['Date'] = pd.date_range(start=start_date, periods=len(df), freq='M')
df.set_index('Date', inplace=True)
df.drop('Time', axis=1, inplace=True)  # Drop the numeric Time column

# Step 4: Visualize the data
plt.figure(figsize=(10, 6))
plt.plot(df['Money'], label='M1 Money Supply')
plt.title('M1 Money Supply (1951-1999)')
plt.xlabel('Date')
plt.ylabel('M1 ($ Billions)')
plt.legend()
plt.show()

# Step 5: Check stationarity with ADF test
def adf_test(series):
    result = adfuller(series)
    print('ADF Statistic:', result[0])
    print('p-value:', result[1])
    print('Critical Values:', result[4])
    if result[1] <= 0.05:
        print("The series is stationary (reject null hypothesis)")
    else:
        print("The series is non-stationary (fail to reject null hypothesis)")

print("ADF Test on Original Series:")
adf_test(df['Money'])

# Take first difference to make it stationary
df['Money_diff'] = df['Money'].diff().dropna()

# Plot the differenced series
plt.figure(figsize=(10, 6))
plt.plot(df['Money_diff'], label='First Difference of M1')
plt.title('First Difference of M1 Money Supply')
plt.xlabel('Date')
plt.ylabel('Difference ($ Billions)')
plt.legend()
plt.show()

print("ADF Test on Differenced Series:")
adf_test(df['Money_diff'].dropna())

# Step 6: Plot ACF and PACF to determine p and q
plt.figure(figsize=(12, 4))
plt.subplot(121)
plot_acf(df['Money_diff'].dropna(), ax=plt.gca(), lags=40)
plt.title('ACF of Differenced Series')

plt.subplot(122)
plot_pacf(df['Money_diff'].dropna(), ax=plt.gca(), lags=40)
plt.title('PACF of Differenced Series')
plt.show()

# Step 7: Fit an ARIMA model (e.g., ARIMA(1,1,1))
# You can adjust (p,d,q) based on ACF/PACF or use auto_arima (optional)
model = ARIMA(df['Money'], order=(1, 1, 1))  # Example: ARIMA(1,1,1)
model_fit = model.fit()

# Summary of the model
print(model_fit.summary())

# Step 8: Plot residuals
residuals = model_fit.resid
plt.figure(figsize=(10, 6))
plt.plot(residuals, label='Residuals')
plt.title('Residuals of ARIMA(1,1,1) Model')
plt.xlabel('Date')
plt.ylabel('Residuals')
plt.legend()
plt.show()

# ACF of residuals
plt.figure(figsize=(6, 4))
plot_acf(residuals, lags=40)
plt.title('ACF of Residuals')
plt.show()

# Step 9: Forecast the next 12 months
forecast = model_fit.forecast(steps=12)
forecast_index = pd.date_range(start=df.index[-1], periods=13, freq='M')[1:]  # Start after last date
plt.figure(figsize=(10, 6))
plt.plot(df['Money'], label='Observed')
plt.plot(forecast_index, forecast, label='Forecast', color='red')
plt.title('M1 Money Supply Forecast')
plt.xlabel('Date')
plt.ylabel('M1 ($ Billions)')
plt.legend()
plt.show()
