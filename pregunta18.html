# Import necessary libraries
import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
from statsmodels.tsa.stattools import adfuller
from statsmodels.tsa.arima.model import ARIMA
from statsmodels.graphics.tsaplots import plot_acf, plot_pacf
from google.colab import files
import io
import re

# Step 1: Upload the text file
print("Please upload your text file containing the macroeconomic data.")
uploaded = files.upload()

# Step 2: Read and parse the file
filename = list(uploaded.keys())[0]
with io.StringIO(uploaded[filename].decode('utf-8')) as f:
    lines = f.readlines()

# Print raw contents for debugging
print("Raw file contents (first 10 lines):")
for i, line in enumerate(lines[:10]):
    print(f"Line {i+1}: '{line.strip()}'")

# Parse the data
data_start = False
year = []
pdi = []
print("\nParsing file contents:")
for line in lines:
    # Check for header line (note the leading spaces)
    if line.strip().startswith("YEAR"):
        data_start = True
        print("Header found:", line.strip())
        continue
    if data_start and line.strip():  # Only process non-empty lines after header
        try:
            # Split on any whitespace (spaces or tabs)
            cols = re.split(r'\s+', line.strip())
            if len(cols) >= 6:  # Ensure enough columns
                year.append(cols[0])  # YEAR (e.g., "1970.1")
                pdi.append(float(cols[2]))  # PDI (3rd column)
                print(f"Parsed: Year={cols[0]}, PDI={cols[2]}")
            else:
                print(f"Skipping malformed line: {line.strip()}")
        except ValueError as e:
            print(f"Error parsing line '{line.strip()}': {e}")

# Check if data was parsed
if not year or not pdi:
    raise ValueError("No data parsed from the file. Check file format or content.")
print(f"Total observations parsed: {len(year)}")

# Create a DataFrame
df = pd.DataFrame({'Year': year, 'PDI': pdi})

# Ensure 'Year' is string type
df['Year'] = df['Year'].astype(str)
print("DataFrame head:\n", df.head())
print("Year column dtype:", df['Year'].dtype)

# Step 3: Convert to time series with quarterly dates
try:
    df['Date'] = pd.to_datetime(df['Year'].str.replace(r'(\d+)\.(\d)', r'\1-Q\2', regex=True))
    df.set_index('Date', inplace=True)
    df.drop('Year', axis=1, inplace=True)
except Exception as e:
    print(f"Date conversion error: {e}")
    raise

# Step 4: Transform PDI to log DPI
df['Log_DPI'] = np.log(df['PDI'])
data_to_model = df['Log_DPI']
label = 'Log DPI'

# Step 5: Visualize the data
plt.figure(figsize=(10, 6))
plt.plot(data_to_model, label=label)
plt.title('Log DPI (1970.1 - 1991.4)')
plt.xlabel('Date')
plt.ylabel('Log DPI (Billions of 1987 $)')
plt.legend()
plt.show()

# Step 6: Check stationarity
def adf_test(series, title):
    result = adfuller(series)
    print(f'ADF Test on {title}:')
    print('ADF Statistic:', result[0])
    print('p-value:', result[1])
    print('Critical Values:', result[4])
    if result[1] <= 0.05:
        print("Stationary (reject null)")
    else:
        print("Non-stationary (fail to reject null)")

adf_test(data_to_model, 'Log DPI')

# First difference
df['Log_DPI_diff'] = data_to_model.diff().dropna()
adf_test(df['Log_DPI_diff'].dropna(), 'Differenced Log DPI')

plt.figure(figsize=(10, 6))
plt.plot(df['Log_DPI_diff'], label='First Difference of Log DPI')
plt.title('First Difference of Log DPI')
plt.xlabel('Date')
plt.ylabel('Difference')
plt.legend()
plt.show()

# Step 7: ACF and PACF
plt.figure(figsize=(12, 4))
plt.subplot(121)
plot_acf(df['Log_DPI_diff'].dropna(), ax=plt.gca(), lags=20)
plt.title('ACF of Differenced Log DPI')

plt.subplot(122)
plot_pacf(df['Log_DPI_diff'].dropna(), ax=plt.gca(), lags=20)
plt.title('PACF of Differenced Log DPI')
plt.show()

# Step 8: Fit ARIMA model
print("Fitting ARIMA model...")
model = ARIMA(data_to_model, order=(1, 1, 1))  # Example ARIMA(1,1,1)
model_fit = model.fit()
print(model_fit.summary())

# Step 9: Diagnostics
residuals = model_fit.resid
plt.figure(figsize=(10, 6))
plt.plot(residuals, label='Residuals')
plt.title('Residuals of ARIMA(1,1,1) Model')
plt.xlabel('Date')
plt.ylabel('Residuals')
plt.legend()
plt.show()

plt.figure(figsize=(6, 4))
plot_acf(residuals, lags=20)
plt.title('ACF of Residuals')
plt.show()

# Step 10: Forecast (next 8 quarters)
forecast = model_fit.forecast(steps=8)
forecast_index = pd.date_range(start=df.index[-1], periods=9, freq='Q')[1:]
plt.figure(figsize=(10, 6))
plt.plot(data_to_model, label='Observed Log DPI')
plt.plot(forecast_index, forecast, label='Forecast', color='red')
plt.title('Log DPI Forecast (1992.1 - 1993.4)')
plt.xlabel('Date')
plt.ylabel('Log DPI')
plt.legend()
plt.show()
