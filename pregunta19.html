# Importar bibliotecas necesarias
import pandas as pd
import matplotlib.pyplot as plt
from statsmodels.tsa.stattools import adfuller, coint
from statsmodels.tsa.vector_ar.var_model import VAR
from google.colab import files

# Paso 1: Subir el archivo .txt
print("Por favor, sube tu archivo .txt con los datos de las tasas de Treasury Bills:")
uploaded = files.upload()

# Paso 2: Leer el archivo subido
filename = list(uploaded.keys())[0]  # Obtener el nombre del archivo subido
with open(filename, 'r') as file:
    lines = file.readlines()

# Paso 3: Parsear los datos
dates = []
gs3m = []
gs6m = []

# Saltar las líneas de encabezado y procesar las filas de datos
for line in lines[9:]:  # Saltar las primeras 9 líneas (ajusta si el encabezado es diferente)
    if line.strip():  # Ignorar líneas vacías
        parts = line.split()  # Separar por espacios en blanco
        if len(parts) == 3:  # Asegurarse de que haya fecha, GS3M y GS6M
            date = parts[0].replace(':', '-') + '-01'  # Convertir '1982:01' a '1982-01-01'
            dates.append(date)
            gs3m.append(float(parts[1]))
            gs6m.append(float(parts[2]))

# Crear un DataFrame
df = pd.DataFrame({
    'Date': pd.to_datetime(dates),
    'GS3M': gs3m,
    'GS6M': gs6m
})
df.set_index('Date', inplace=True)

# a. Graficar las dos series temporales en el mismo diagrama
plt.figure(figsize=(12, 6))
plt.plot(df.index, df['GS3M'], label='Tasa de 3 meses (GS3M)', color='blue')
plt.plot(df.index, df['GS6M'], label='Tasa de 6 meses (GS6M)', color='red')
plt.title('Tasas de Treasury Bills de 3 y 6 meses (Ene 1982 - Jun 2001)')
plt.xlabel('Fecha')
plt.ylabel('Tasa de Interés (%)')
plt.legend()
plt.grid()
plt.show()

print("Observación: Las dos series parecen moverse juntas a lo largo del tiempo, lo que sugiere una posible relación a largo plazo. Ambas muestran tendencias y fluctuaciones, con mayor volatilidad a principios de los 80 y una disminución general hacia 2001.")

# b. Análisis formal de raíz unitaria (Prueba de Dickey-Fuller Aumentada)
def adf_test(series, name):
    result = adfuller(series, autolag='AIC')
    print(f'\nPrueba ADF para {name}:')
    print(f'Estadístico ADF: {result[0]}')
    print(f'p-valor: {result[1]}')
    print(f'Valores Críticos: {result[4]}')
    if result[1] < 0.05:
        print(f"{name} es estacionaria (rechazamos la hipótesis nula de raíz unitaria)")
    else:
        print(f"{name} es no estacionaria (no rechazamos la hipótesis nula de raíz unitaria)")

adf_test(df['GS3M'], 'GS3M')
adf_test(df['GS6M'], 'GS6M')

# c. Prueba de cointegración
coint_test = coint(df['GS3M'], df['GS6M'])
print('\nPrueba de Cointegración (Engle-Granger):')
print(f'Estadístico de la prueba: {coint_test[0]}')
print(f'p-valor: {coint_test[1]}')
print(f'Valores Críticos: {coint_test[2]}')
if coint_test[1] < 0.05:
    print("Las series están cointegradas (rechazamos la hipótesis nula de no cointegración)")
else:
    print("Las series no están cointegradas (no rechazamos la hipótesis nula de no cointegración)")

# d. Significado económico de la cointegración
print("\nSignificado Económico de la Cointegración:")
print("Si GS3M y GS6M están cointegradas, implica una relación estable a largo plazo entre las tasas de 3 y 6 meses. Esto podría reflejar expectativas del mercado o la dinámica de la estructura temporal de tasas, donde las tasas a corto plazo se ajustan para alinearse con las de mayor plazo.")
print("Si no están cointegradas, sugiere que las dos tasas pueden divergir sin un equilibrio estable, posiblemente debido a fuerzas económicas distintas o ineficiencias del mercado que afectan cada vencimiento de manera independiente.")

# e. Consideración para el modelo VAR
print("\nConsideración para el Modelo VAR:")
print("Para estimar un modelo VAR con cuatro rezagos, necesitamos verificar la estacionariedad:")
print("- Si las series son no estacionarias pero cointegradas, puedes estimar el VAR en niveles (o usar un VECM).")
print("- Si las series son no estacionarias y no cointegradas, debes usar primeras diferencias para asegurar estacionariedad.")
print("- Si las series son estacionarias, puedes usar niveles directamente.")

# Verificar estacionariedad de las primeras diferencias
df_diff = df.diff().dropna()
adf_test(df_diff['GS3M'], 'GS3M (Primera Diferencia)')
adf_test(df_diff['GS6M'], 'GS6M (Primera Diferencia)')

print("\nBasado en las pruebas ADF anteriores:")
print("Si las series originales son no estacionarias (p > 0.05) y las primeras diferencias son estacionarias (p < 0.05), usa primeras diferencias para el VAR.")
print("Si las series originales están cointegradas, podrías usar niveles o un VECM. De lo contrario, se recomienda diferenciar.")

# Ejemplo de modelo VAR (en primeras diferencias, asumiendo no estacionariedad)
model = VAR(df_diff)
results = model.fit(4)  # 4 rezagos
print("\nResumen del Modelo VAR (Primeras Diferencias, 4 rezagos):")
print(results.summary())
